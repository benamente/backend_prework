class Array
  def print_with(data)
    self.each do |x|
      x.print_my()
      max_length = @self.max_attribute("length")
      num_spaces = max_length - x.cryp_text.length + 2
      print " " * num_spaces
      data.each do |d|
        if x.public_send(d).is_a? Array
          print x.public_send(d).to_uncluttered_string_limited(40)
        else
          print x.public_send(d).to_s
        end
        #print "#{x.x_string.d}"
      end
      puts ""
    end
  end
end
